function showSection(id) {

  document.querySelectorAll('.section').forEach(s => s.classList.remove('active'));

  document.getElementById(id).classList.add('active');

}

  en: {

    teamMembersText: "[Add all team members]",

  }

};


function setLang(lang) {

  const t = T[lang];

  if (!t) return;


  document.getElementById('subtitle-text').textContent = t.subtitle;

  document.getElementById('nav-home').textContent = t.navHome;

  document.getElementById('nav-team').textContent = t.navTeam;

  document.getElementById('home-heading').textContent = t.homeHeading;

  document.getElementById('team-heading').textContent = t.teamHeading;

  document.getElementById('team-members').textContent = t.teamMembers;

  document.getElementById('team-members-text').textContent = t.teamMembersText;

  document.getElementById('footer-text').textContent = t.footer;

}


setLang('en');
